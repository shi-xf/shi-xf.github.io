2.事件绑定

$(‘div’).on(“click”,function(){})
$(‘div’).on(“click mouseover”,function(){}) 绑定两个事件，同一个函数
$(‘div’).on({
“click”:function(){},
“mouseover”:function(){} 绑定两个事件，不同的函数
})
事件委托
$(“ul”).on(“click”,”li”,function(){}) //里面的this指向事件源(li这个元素)
事件移除
$(“div”).off() 没有参数时会把所有的事件移除
$(“div”).off(“click”) 移除所有的点击事件
$(‘div’).on(“click.aa”,function(){}) 自定义添加名字以便移除
$(‘div’).on(“click.bb”,function(){})
$(“div”).off(“click.bb”) 移除click.bb点击事件 .bb是通过命名空间的形式给点击事件起的名字
one事件，只执行一次
$(“div”).one(“click”,function(){})

3.trigger

触发默认的事件，例如：表单中的提交，重置事件
$(“form”).trigger(“submit”)
触发自定义事件：先通过on自定义一个事件，然后通过元素.trigger(“自定义事件名”)触发自定义事件
$(“div”).on(“aa”,function(){})
$(“div”).trigger(“aa”) //触发自定义事件

5.toggle

$(“div”).toggle(function(){},function(){}) //函数点击依次执行

6.隐式循环，链式调用

7.事件对象

$(“div”).click(function(ev){
ev.originalEvent //原生的事件对象
ev.pageX;ev.pageY //获取鼠标到页面的x轴和y轴的距离
ev.clientX;ev.clientY //获取鼠标到可视窗口的x轴和y轴的距离
ev.offsetX;ev.offsetY //获取鼠标到绑定事件的元素的x轴和y轴的距离
ev.screenX;ev.screenY //获取鼠标到屏幕的x轴和y轴的距离
})
ev.delegateTarget //绑定事件的元素 dom对象（事件委托的父元素）
ev.currentTarget //事件源 dom对象（事件委托的子元素）
ev.target //事件源 dom对象（事件委托的子元素）
ev.which //1表示左键 2表示滚轮 3表示右键
$(document).keydown(function(ev){
ev.which //获取键盘的键码值
ev.ctrlKey //检测有无按下ctrl键
ev.altKey //检测有无按下alt键
ev.shiftKey//检测有无按下shift键
})
ev.preventDefault() //阻止事件默认行为
return false //阻止事件默认行为
ev.stopPropagation() //阻止事件冒泡
return false //阻止事件冒泡

8.动画

$(“div”).show(); //显示 show(1000,function(){}) 1s后显示，第二个参数为函数
$(“div”).hide(); //隐藏
$(“div”).toggle() //显示隐藏
$(“div”).fadeIn(); //显示（透明度变化）
$(“div”).fadeOut(); //隐藏（透明度变化）
$(“div”).fadeTo(1000,0.3); //指定透明程度，1s后透明到0.3
$(“div”).fadeToggle(); //显示隐藏
$(“div”).slideDown() //高度变化，向下显示
$(“div”).slideUp() //高度变化，向上隐藏
$(“div”).slideToggle() //向下显示,向上隐藏

9.stop

$(“div”).finish() //动画完成，会直接到达最后的状态
$(“div”).stop() //动画停止，第一个参数表示是否清空动画队列 true清空动画队列，第二个参数表示是否立即完成动画 true立即完成动画
$(“div”).delay().slideUp() //延迟之后才执行

10.自定义动画

元素.animate({“属性名”:”属性值”},过渡时长,动画方式,函数)
$(“div”).animate({“width”:300,”height”:300},1000,”linear”,function(){})
元素.animate({“属性名”:”属性值”},{duration:3000,queue:false})
当queue的值指定为false时，动画队列的动画会一块执行

11.动画队列中的queue

$(“div”).slideUp(2000).queue(function(aa){
$(“div”).css(“background”:”yellow”);
aa(); // 确保queue后面的动画能正常执行
}).slideDown(2000)
queue()是一个插队的函数，在函数中可以指定自定义的样式的变化，需要调用queue参数function中形参，这样是为了确保queue后面的动画能正常执行
